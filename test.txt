sub sumTo(num) {
        sum = 0
        while (num > 0) {
            sum = (sum + num)
            num = (num ~ 1)
        }
        print(sum)
    }
call sumTo(10)
call sumTo((10*10))

sub stamp(word, times) {
        final = "" 
        while (times > 0) {
            final = (final + word)
            times = (times~1)
        }
        print(final)
    }
call stamp("foo", 3)
call stamp(("X"+"Y"), (4*2))
